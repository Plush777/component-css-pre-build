$('.btnHeadToggle').on({
    "click":function (){
        $(this).toggleClass('active').siblings('.btnHeadToggle').removeClass('active');
        if(!$(this).hasClass('active')){
            $('#header').removeClass('active');
        }else if($(this).hasClass('active')){
            $('#header').addClass('active');
        }
    }
});
