                    }
                }

        }
    }
}

$('.tabList li').on({
  "click":function(){
    $(this).addClass('active').siblings('li').removeClass('active');
    $(this).closest('.tabNav').siblings('.tabCont').eq($(this).index()).addClass('active').siblings('.tabCont').removeClass('active');
  }
})
